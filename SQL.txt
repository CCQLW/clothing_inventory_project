CREATE DATABASE clothing_inventory;
use clothing_inventory

CREATE TABLE GRN
(
id INT,
receipt_number VARCHAR(16),
warehouse VARCHAR(50),
storage_time DATE,
agent VARCHAR(50),
source VARCHAR(50),
put_delete BINARY
);

CREATE TABLE warehousing_detail
(
id INT,
article_number VARCHAR(50),
trade_name VARCHAR(50),
color_no VARCHAR(50),
size INT,
number INT,
put_delete BINARY
);
